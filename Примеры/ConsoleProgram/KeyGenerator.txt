    public partial class Form1 : Form

        const string listKey = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";// Список символов
        Random rnd = new Random();// Случайность
        
        string GenKey()
        {
            string partKey = string.Empty;// Пустое 
            
            for (int i = 0; i < 5; i++)// Цикл создания 5 символов
            {
                partKey += listKey[rnd.Next(0,listKey.Length)];
            }
            return partKey;//Возврат результата
                
        }

        {
            InitializeComponent();
        }

            }
