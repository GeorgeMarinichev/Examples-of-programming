public static class StringExtension
{
    public static int CharCount(this string str, char c)//Специальный класс и метод расширения
    {
        int counter = 0;

        for(int i = 0; i < str.Length; i++)
        {
            if(str[i] == c)
            {
                counter++;
            }
        }
        return counter;
    }
}
