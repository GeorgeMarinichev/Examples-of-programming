namespace Genesis; // Пространство имен

class Person // Класс хранилище
{
    public const string TYPE = "Person";
    public readonly string name = "Undifined";
    static int counter = 0;// Поле
    public int age = 0;// Поле
    static int retirementAge = 65;// Поле
    public static int RetirementAge // Свойства
    {
        get { return retirementAge; }
        //set { retirementAge = value; }
        //set { if (value > 1 $$ value < 100 ) retirementAge = value; }
    }
    
    static Person() // Конструкор static
    {
        Console.WriteLine("Конструкор static");
    }

    public Person()// Конструкор пустой
    {
        counter++;
        Console.WriteLine($"Вызов конструктора класса {TYPE}: {name} №{counter}:");
    }
    
    public Person(int age) // Конструкор 1
    {
        this.age = age;
    }

    public Person(int age, string name) // Конструкор 1
    {
        this.age = age;
        this.name = name;
    }

    public void CheckAge()// CheckAge
    {
        if(age > retirementAge)
        {
            Console.WriteLine($"Возраст:{age} Уже на пенсии");
        }
        else if(age < retirementAge)
        {
            Console.WriteLine($"Возраст:{age} До пенсии осталось: {retirementAge - age}");
        }
    }
}
