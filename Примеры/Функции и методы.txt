using System;

    internal class Program

        static int myMetod(int[] massiv, int val)// Метод поиска чиисла в массиве
        {
            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("Поиск числа...)");
            for (int i = 0; i < massiv.Length; i++)
            {
                if (massiv[i] == val)
                {
                    Console.WriteLine($"Есть такое число:{i}");
                    return i;
                }
            }
            Console.WriteLine("Нет такого числа:");
            return -1;
        }

        static int[] myMassiv(uint lenght, int min_val, int max_val)//Метод генератор одномерного массива
        {
            int[] massiv = new int[lenght];
            Random rand = new Random();
            
            
            for (int i = 0; i < massiv.Length; i++)
            {
                massiv[i] = rand.Next(min_val, max_val);
            }
            Console.ForegroundColor = ConsoleColor.Cyan;
            for (int i = 0; i < massiv.Length; i++)
            {
                Console.Write(massiv[i] + "\t");
            }
            Console.WriteLine();
            return massiv;
        }

            uint mass_x;
            int data_x, data_y, val;
            mass_x = 0;
            val = 0;
            data_x = 0;
            data_y = 0;
            Random rand = new Random();

            try
            {
                Console.Write("Введите длину массива: ");
                mass_x = uint.Parse(Console.ReadLine());
                Console.Write("Введите диапазон значения от: ");
                data_x = int.Parse(Console.ReadLine());
                Console.Write("Введите диапазон значения до: ");
                data_y = int.Parse(Console.ReadLine());
                Console.Write("Введите число поиска: ");
                val = int.Parse(Console.ReadLine());
            }
            catch (Exception)
            {
                Console.ForegroundColor = ConsoleColor.Red;
                Console.WriteLine(" Не верное значение!!!");
                Console.ReadKey();
                //throw;
            }
            
            myMetod(myMassiv(mass_x, data_x, data_y), val);
            
            //Console.WriteLine();
