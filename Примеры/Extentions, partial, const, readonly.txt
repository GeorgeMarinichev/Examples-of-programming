namespace Genesis; // Пространство имен

partial class Person // Класс partial(частичный)
{
    public const string PER = "Person"; // Поле константа переменная
    public readonly string _pr = "1985"; // Поле readonly

    public Person() {} // Пустой конструктор
    public Person( string FirstName, string LastName) // Конструктор
    {
        this.FirstName = FirstName;
        this.LastName = LastName;
    }

    public string FirstName {get; set;}// Свойства
    public string LastName {get; set;}// Свойства

    public partial string GetFullName() // Метод с реализацией partial(частичный)
    {
        return FirstName + " " + LastName;
    }

}

{

        WriteLine(dateTime);

    }
